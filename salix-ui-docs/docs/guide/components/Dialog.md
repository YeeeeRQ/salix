## 示例2 调用showDialog

<Button @click="showDialog">show dialog</Button>

<script setup>
import { ref } from "vue";
import { Button, Dialog, openDialog } from "salix-ui";

const visible = ref(false);
const handleBtnClick = () => {
    visible.value = !visible.value;
};
const f1 = () => {
    console.log("call f1");
    return true;
};
const f2 = () => {
    console.log("call f2");
    return true;
};

const showDialog = () => {
    openDialog({
    title: () => "标题",
    content: () => "你好",
    ok() {
        console.log("ok");
        return true;
    },
    cancel() {
        console.log("cancel");
        return true;
    },
    });
};
</script>
